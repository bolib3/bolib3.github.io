\begin{align*}
  &\minimise_{x_1, x_2, x_3, x_4} \quad
  && (x_1 + x_2 - 2)^2 + (x_3 + x_4 - 2)^2 \\
  &\subjectto
  && h_1 = 1.2 \\
  &&& h_2 = 6 \\
  &&& x_1 - x_2 = 3 \\
  &&& h_1 = \min_{u_1, u_2, u_3, u_4}
  \left\{
  \begin{aligned}
  &\sum_{i=1}^4(u_i - x_i)^2  \\
  &\textrm{s.t. } 3u_1 + u_2 + 2u_3 + u_4 = 6
  \end{aligned}
  \right.\\
  &&& h_2 = \min_{u_1, u_2, u_3, u_4}
  \left\{
  \begin{aligned}
  &\sum_{i=1}^4(u_i - x_i)^2  \\
  &\textrm{s.t. } u_1 + u_2 + u_3 + 2u_4 = 7.
  \end{aligned}
  \right.
  \end{align*}
